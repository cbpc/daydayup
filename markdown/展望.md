# 函数式编程
```typescript
import * as R from 'ramda';
export interface IArrFilter { 
    key: string;
    data: any    
 };
let getDataByIdx: IArrFilter => Array<any> = ({ key, data }) =>
  R.pluck(key)(data);

let getUniqByIdx: IArrFilter => Array<any> = ({ key, data }) =>
  R.uniq(
    getDataByIdx({
      key,
      data,
    })
  );
```
